def generate_compliance_report(urls):

        'licensed_content': 0,
        'training_allowed': 0,
        'attribution_required': 0,

            if result.restrictions.attribution_required:

            if not result.restrictions.allow_commercial:
                report['commercial_restricted'] += 1
    
    return report
